-- create table authors in database Liprary in phpmyadmin (2 columns)
CREATE TABLE Authors
(
    author_id INT
    AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR
    (15) NOT NULL
);

    -- create table books in database Liprary in phpmyadmin (3 columns)
    CREATE TABLE Books
    (
        book_id INT
        AUTO_INCREMENT PRIMARY KEY,
        title VARCHAR
        (15) NOT NULL,
        author_id INT,
        FOREIGN KEY
        (author_id) REFERENCES Authors
        (author_id)
    );






        -- task 2
        -- create database Shop in phpmyadmin

        -- create table Customers in database Shop in phpmyadmin (3 columns)
        CREATE TABLE Customers
        (
            customer_id INT
            AUTO_INCREMENT PRIMARY KEY,
             customer_name VARCHAR
            (50) NOT NULL,
              order_id INT UNIQUE, //one
              FOREIGN KEY
            (order_id) REFERENCES Orders
            (order_id)
        );

     -- create table Orders in database Shop in phpmyadmin (2 columns)
    CREATE TABLE Orders
    (
        order_id INT
        AUTO_INCREMENT PRIMARY KEY,
        customer_id INT , // many
         FOREIGN KEY
        (customer_id) REFERENCES Customers
        (customer_id)
   );

    -- create table Products in database Shop in phpmyadmin (2 columns)
          CREATE TABLE Products
          (
              product_id INT
              AUTO_INCREMENT PRIMARY KEY,
               product_name VARCHAR
              (50) NOT NULL,
               price DECIMAL
              (10, 2) NOT NULL  
       );

    -- create table OrdersProducts in database Shop in phpmyadmin (2 columns referance) junction table
    CREATE TABLE OrdersProducts
     (
        order_id INT,
         product_id INT,
         FOREIGN KEY (order_id) REFERENCES Orders(order_id),
         FOREIGN KEY (product_id) REFERENCES Products(product_id)
     );

-- Primary Key :  Only one per table.    Never allows NULL. Every row must have a value.    Uniquely identifies a record (for joins/references).    Usually creates a Clustered Index.
-- Unique Constraint : Multiple allowed per table.    Allows one NULL (in most databases like SQL Server/MySQL).*    Prevents duplicates in non-primary columns (e.g., Email).    Usually creates a Non-clustered Index.
