-- display all orders with user details
SELECT * FROM Orders 
JOIN USERS ON Orders.user_id = Users.user_id;

-- display all pending orders
SELECT order_id, total_price, order_status 
FROM Orders 
WHERE order_status = 'Pending';

-- display product names and prices of products priced above 100 with warranty
SELECT product_name, price 
FROM Products 
WHERE price > 100 AND has_warranty = TRUE;

-- display first and last names of all users ordered by last name ascending
SELECT first_name, last_name 
FROM Users 
ORDER BY last_name ASC;

-- display total number of delivered orders
SELECT COUNT(order_id) AS total_delivered_orders 
FROM Orders 
WHERE order_status = 'Delivered';

-- display total revenue from all non-cancelled orders
SELECT SUM(total_price) AS total_revenue 
FROM Orders 
WHERE order_status != 'Cancelled';

-- display average price of products in category_id = 1
SELECT AVG(price) AS average_product_price 
FROM Products 
WHERE category_id = 1;
